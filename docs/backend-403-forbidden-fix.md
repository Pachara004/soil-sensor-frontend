### **1. ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç authMiddleware ‡πÉ‡∏´‡πâ‡∏£‡∏≠‡∏á‡∏£‡∏±‡∏ö Firebase Token:**

#### **A. ‡∏õ‡∏±‡∏ç‡∏´‡∏≤‡πÄ‡∏î‡∏¥‡∏°:**

const authMiddleware = async (req, res, next) => {
  try {
    const token = req.header('Authorization')?.replace('Bearer ', '');
    
    if (!token) {
      return res.status(401).json({ message: 'No token' });
    }

#### **B. ‡∏Å‡∏≤‡∏£‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç:**
```javascript
// ‚úÖ ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç: ‡∏£‡∏≠‡∏á‡∏£‡∏±‡∏ö‡∏ó‡∏±‡πâ‡∏á Firebase token ‡πÅ‡∏•‡∏∞ JWT token
const admin = require('firebase-admin');

const authMiddleware = async (req, res, next) => {
  try {
    const token = req.header('Authorization')?.replace('Bearer ', '');
    
    if (!token) {
      return res.status(401).json({ message: 'No token' });
    }
    
    let decoded;
    
    try {
      // ‚úÖ ‡∏•‡∏≠‡∏á‡πÉ‡∏ä‡πâ Firebase Admin SDK ‡∏Å‡πà‡∏≠‡∏ô
      decoded = await admin.auth().verifyIdToken(token);
      
      // ‚úÖ ‡∏î‡∏∂‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• user ‡∏à‡∏≤‡∏Å PostgreSQL ‡πÇ‡∏î‡∏¢‡πÉ‡∏ä‡πâ Firebase UID
      const userQuery = await db.query(`
        SELECT userid, user_name, user_email, role, firebase_uid
        FROM users 
        WHERE firebase_uid = $1 OR user_email = $2
      `, [decoded.uid, decoded.email]);
      
      if (userQuery.rows.length === 0) {
        return res.status(401).json({ message: 'User not found in database' });
      }
      
      req.user = userQuery.rows[0];
      console.log('‚úÖ Firebase token verified, user:', req.user);
      
    } catch (firebaseError) {
      console.log('‚ö†Ô∏è Firebase token verification failed, trying JWT...');
      
      try {
        // ‚úÖ Fallback: ‡∏•‡∏≠‡∏á‡πÉ‡∏ä‡πâ JWT token
        decoded = jwt.verify(token, process.env.JWT_SECRET);
        
        // ‚úÖ ‡∏î‡∏∂‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• user ‡∏à‡∏≤‡∏Å PostgreSQL ‡πÇ‡∏î‡∏¢‡πÉ‡∏ä‡πâ userid
        const userQuery = await db.query(`
          SELECT userid, user_name, user_email, role, firebase_uid
          FROM users 
          WHERE userid = $1
        `, [decoded.userid]);
        
        if (userQuery.rows.length === 0) {
          return res.status(401).json({ message: 'User not found in database' });
        }
        
        req.user = userQuery.rows[0];
        console.log('‚úÖ JWT token verified, user:', req.user);
        
      } catch (jwtError) {
        console.error('‚ùå Both Firebase and JWT token verification failed');
        return res.status(401).json({ message: 'Invalid token' });
      }
    }
    
    next();
  } catch (error) {
    console.error('‚ùå Auth middleware error:', error);
    res.status(401).json({ message: 'Authentication failed' });
  }
};
```

### **2. ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç adminMiddleware:**

#### **A. ‡∏õ‡∏±‡∏ç‡∏´‡∏≤‡πÄ‡∏î‡∏¥‡∏°:**
```javascript
// ‚ùå ‡∏õ‡∏±‡∏ç‡∏´‡∏≤: adminMiddleware ‡πÑ‡∏°‡πà‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö role ‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á
const adminMiddleware = (req, res, next) => {
  if (req.user.role !== 'admin') {
    return res.status(403).json({ 
      message: 'Access denied. Admin required.',
      userRole: req.user.role 
    });
  }
  next();
};
```

#### **B. ‡∏Å‡∏≤‡∏£‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç:**

// ‚úÖ ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç: ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö role ‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á
const adminMiddleware = (req, res, next) => {
  console.log('üîç Checking admin role for user:', req.user);
  
  if (!req.user) {
    return res.status(401).json({ message: 'User not authenticated' });
  }
  
  // ‚úÖ ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö role ‡πÉ‡∏ô‡∏´‡∏•‡∏≤‡∏¢‡∏£‡∏π‡∏õ‡πÅ‡∏ö‡∏ö

    console.log('‚ùå Access denied. User role:', userRole);
    return res.status(403).json({ 
      message: 'Access denied. Admin required.',
      userRole: userRole,
      userId: req.user.userid,
      userName: req.user.user_name
    });
  }
  
  console.log('‚úÖ Admin access granted for user:', req.user.user_name);
  next();
};

### **3. ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç API Endpoints:**

#### **A. GET /api/reports:**
```javascript
// ‚úÖ ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç: ‡πÉ‡∏ä‡πâ authMiddleware ‡πÅ‡∏•‡∏∞ adminMiddleware ‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á
app.get('/api/reports', authMiddleware, adminMiddleware, async (req, res) => {
  try {
    console.log('üìä Admin requesting reports:', req.user.user_name);
    
    const { status = 'new', limit = 20, offset = 0 } = req.query;
    
    // ‚úÖ ‡∏î‡∏∂‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• reports ‡∏û‡∏£‡πâ‡∏≠‡∏°‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• user ‡πÅ‡∏•‡∏∞ admin
    const query = `
      SELECT 
        r.*,
        u.user_name,
        u.user_email,
        a.user_name as admin_name
      FROM reports r
      LEFT JOIN users u ON r.user_id = u.userid
      LEFT JOIN users a ON r.admin_id = a.userid
      WHERE r.status = $1
      ORDER BY r.created_at DESC
      LIMIT $2 OFFSET $3
    `;
    
    const result = await db.query(query, [status, limit, offset]);
    
    // ‚úÖ ‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏ó‡∏±‡πâ‡∏á‡∏´‡∏°‡∏î
    const countResult = await db.query(
      'SELECT COUNT(*) as total FROM reports WHERE status = $1',
      [status]
    );
    
    const total = parseInt(countResult.rows[0].total);
    
    console.log(`‚úÖ Found ${result.rows.length} reports for admin`);
    
    res.json({
      reports: result.rows,
      pagination: {
        total,
        limit: parseInt(limit),
        offset: parseInt(offset),
        hasMore: (parseInt(offset) + parseInt(limit)) < total
      }
    });
    
  } catch (error) {
    console.error('‚ùå Error fetching reports:', error);
    res.status(500).json({ error: 'Failed to fetch reports' });
  }
});
```

#### **B. GET /api/admin/devices:**
```javascript
// ‚úÖ ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç: ‡πÉ‡∏ä‡πâ authMiddleware ‡πÅ‡∏•‡∏∞ adminMiddleware ‡∏≠‡∏¢‡πà‡∏≤‡∏á‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á
app.get('/api/admin/devices', authMiddleware, adminMiddleware, async (req, res) => {
  try {
    console.log('üì± Admin requesting devices:', req.user.user_name);
    
    const { limit = 50, offset = 0, search = '' } = req.query;
    
    // ‚úÖ ‡∏™‡∏£‡πâ‡∏≤‡∏á query ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏î‡∏∂‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• devices ‡∏û‡∏£‡πâ‡∏≠‡∏°‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• user
    let query = `
      SELECT 
        d.deviceid,
        d.device_name,
        d.device_type,
        d.device_status,
        d.created_at,
        d.updated_at,
        u.user_name,
        u.user_email,
        COUNT(m.id) as measurement_count
      FROM devices d
      LEFT JOIN users u ON d.user_id = u.userid
      LEFT JOIN measurements m ON d.deviceid = m.deviceid
    `;
    
    const conditions = [];
    const params = [];
    let paramIndex = 1;
    
    // ‚úÖ ‡πÄ‡∏û‡∏¥‡πà‡∏°‡πÄ‡∏á‡∏∑‡πà‡∏≠‡∏ô‡πÑ‡∏Ç‡∏Å‡∏≤‡∏£‡∏Ñ‡πâ‡∏ô‡∏´‡∏≤
    if (search) {
      conditions.push(`d.device_name ILIKE $${paramIndex}`);
      params.push(`%${search}%`);
      paramIndex++;
    }
    
    if (conditions.length > 0) {
      query += ` WHERE ${conditions.join(' AND ')}`;
    }
    
    query += ` GROUP BY d.deviceid, d.device_name, d.device_type, d.device_status, d.created_at, d.updated_at, u.user_name, u.user_email`;
    query += ` ORDER BY d.deviceid ASC`;
    query += ` LIMIT $${paramIndex} OFFSET $${paramIndex + 1}`;
    
    params.push(limit, offset);
    
    const result = await db.query(query, params);
    
    // ‚úÖ ‡∏ô‡∏±‡∏ö‡∏à‡∏≥‡∏ô‡∏ß‡∏ô‡∏ó‡∏±‡πâ‡∏á‡∏´‡∏°‡∏î
    let countQuery = `
      SELECT COUNT(DISTINCT d.deviceid) as total
      FROM devices d
      LEFT JOIN users u ON d.user_id = u.userid
    `;
    
    if (conditions.length > 0) {
      countQuery += ` WHERE ${conditions.join(' AND ')}`;
    }
    
    const countResult = await db.query(countQuery, params.slice(0, -2));
    const total = parseInt(countResult.rows[0].total);
    
    console.log(`‚úÖ Found ${result.rows.length} devices for admin`);
    
    res.json({
      devices: result.rows,
      pagination: {
        total,
        limit: parseInt(limit),
        offset: parseInt(offset),
        hasMore: (parseInt(offset) + parseInt(limit)) < total
      }
    });
    
  } catch (error) {
    console.error('‚ùå Error fetching admin devices:', error);
    res.status(500).json({ error: 'Failed to fetch devices' });
  }
});
```
